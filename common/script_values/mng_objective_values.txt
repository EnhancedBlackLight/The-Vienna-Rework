aver_rich_pop_wealth = {
	every_scope_pop = {
		limit = { strata = rich }
		add = {
			add = standard_of_living
		}
	}
	divide = {
		every_scope_pop = {
			limit = { strata = rich }
			add = 1
		}
	}
}

	value = 0
	every_scope_state = {
		every_scope_building = {
			limit = { 
				is_government_funded = no 
				is_subsidized = no
				is_subsistence_building = no
			}
			add = weekly_profit			
		}
	}
	divide = 1000
}
national_inflation_value = {
	if = {
		limit = { has_variable = national_inflation }
		add = root.var:national_inflation
	}
}

country_inflation_level = {
	value = { #铸币数量
		if = {
			limit = {
				capital = {
					any_scope_building = {
						is_building_type = building_center_bank
						has_active_production_method = pm_expansionary_monetary_policy
					}
				}
			}
			add = root.gdp
			divide = 1000000
			multiply = 750
		}
		if = {
			limit = {
				capital = {
					any_scope_building = {
						is_building_type = building_center_bank
						has_active_production_method = pm_no_monetary_policy
					}
				}
			}
			add = root.gdp
			divide = 1000000
			multiply = 500
		}
		if = {
			limit = {
				capital = {
					any_scope_building = {
						is_building_type = building_center_bank
						has_active_production_method = pm_tight_monetary_policy
					}
				}
			}
			add = root.gdp
			divide = 1000000
			multiply = 250
		}
	}
	divide = {
		add = root.income
		multiply = 500
		min = 1
	}
	if = {
		limit = {
			capital = {
				any_scope_building = {
					is_building_type = building_center_bank
					has_active_production_method = pm_tight_monetary_policy
				}
			}
		}
		add = -0.0002
	}
}
