je_peru_bolivia = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = peru_bolivia_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		trigger_event = {
			id = south_america.1
		}
	}

	fail = {
		AND = {
			OR = {
				NOT = { exists = c:PEU }
				c:PEU = {
					NOT = { is_subject_of = ROOT }
				}
			}
			OR = {
				NOT = { exists = c:NEU }
				c:NEU = {
					NOT = { is_subject_of = ROOT }
				}
			}		
		}

	}

	immediate = {
        set_variable = {
			name = peru_bolivia_var
			value = 0
		}
    }

	current_value = {
		value = root.var:peru_bolivia_var
	}

	goal_add_value = {
		value = 120
	}

	progressbar = yes

	weight = 10000
}
