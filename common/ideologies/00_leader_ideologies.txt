		always = yes
		
		scope:interest_group = {
			NOR = {
				has_variable = communist_ig_var
				has_variable = chose_vanguardism
				has_variable = chose_fascism
				has_variable = chose_conservatism
				has_variable = chose_ethno_nationalism
				has_variable = chose_marketanarchy #NEW
				has_variable = chose_marketoligarchy #NEW
				has_variable = chose_marketliberalism #NEW
			}
		}
	}
	
	leader_weight = {
		value = 300	
		# angry IG's more likely to have opinions
		if = {
			limit = {
				scope:interest_group = {
					ig_approval < -5
				}
			}
			add = -200
		}
		# what makes a man turn neutral? lust for gold? power? or were you just born with a heart full of neutrality?
		if = {
			limit = {
				scope:interest_group = {
					ig_approval = 0
				}
			}
			add = 1
		}
		if = { #makes moderates less likely as more society techs related to politics are researched
			limit = {
				has_technology_researched = nationalism
			}
			multiply = 0.9
		}
		if = {
			limit = {
				has_technology_researched = egalitarianism
			}
			multiply = 0.9
		}
		if = {
			limit = {
				has_technology_researched = labor_movement
			}
			multiply = 0.9
		}
		if = {
			limit = {
				has_technology_researched = socialism
			}
			multiply = 0.9
		}
		if = {
			limit = {
				has_technology_researched = philosophical_pragmatism
			}
			multiply = 0.9
		}
		if = {
			limit = {
				has_technology_researched = political_agitation
			}
			multiply = 0.9
		}
	}
}

				owner = {

		}
