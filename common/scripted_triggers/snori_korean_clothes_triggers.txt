snori_character_westernize_trigger = {
	exists = scope:character
	scope:character = {
		owner = {
			has_modifier = snori_kor_clothes_modifier	
		}
	}
}


snori_pop_westernize_trigger = {
	exists = scope:pop
	scope:pop = {
		owner = {
			has_modifier = snori_kor_clothes_modifier
		}
	}
}


snori_korean_character_trigger = {
	exists = scope:character
	scope:character = {
		owner = {
			culture = cu:korean
		}
	}
}
