restore_order_korea = {
	is_shown = {
		exists = c:JAP
		root = c:JAP
		NOT = { has_variable = korea_restore_order }
	}

    possible = {
		has_technology_researched = nationalism
	}

    when_taken = {
        if = {
			limit = {
				owns_entire_state_region = STATE_BUSAN
				owns_entire_state_region = STATE_YANGHO
				owns_entire_state_region = STATE_SEOUL
				owns_entire_state_region = STATE_PYONGYANG
				owns_entire_state_region = STATE_SARIWON
			}
			trigger_event = { id = dvg_manifest_destiny_japan.1 }
		}
		else = {
			add_journal_entry = {
				type = je_restore_order_korea
			}
		}
		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
				multiplier = -1
			}
		}
		# Add Claims
		s:STATE_BUSAN = { 
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							owner = ROOT
						}
					}
				}
				add_claim = ROOT 
			}
		}
		s:STATE_YANGHO = { 
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							owner = ROOT
						}
					}
				}
				add_claim = ROOT 
			}
		}
		s:STATE_SEOUL = { 
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							owner = ROOT
						}
					}
				}
				add_claim = ROOT 
			}
		}
        s:STATE_PYONGYANG = { 
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							owner = ROOT
						}
					}
				}
				add_claim = ROOT 
			}
		}
        s:STATE_SARIWON = { 
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							owner = ROOT
						}
					}
				}
				add_claim = ROOT 
			}
		}
		set_variable = korea_restore_order
    }

	ai_chance = {
		base = 10
	}
}
